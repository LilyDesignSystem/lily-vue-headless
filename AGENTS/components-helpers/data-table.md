```tsx
<DataTable>
    <DataTableHead>
        <DataTableTR>

    </DataTableHead>
    <DataTableBody>
        <DataTableTR>

    </DataTableBody>
    <DataTableFoot>
        <DataTableTR>
            <DataTableTH>…</DataTableTH>

    </DataTableFoot>
```
